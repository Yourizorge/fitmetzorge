import { serve } from "https://deno.land/std@0.224.0/http/server.ts"; import { createClient } from "https://esm.sh/@supabase/supabase-js@2.45.4"; const WEBAPP_URL = "https://appfmz.nl"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS" }; function jsonResponse(body: Record, status = 200) { return new Response(JSON.stringify(body), { status, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } function cleanEmail(value: unknown) { return String(value || "").trim().toLowerCase(); } function safeWebappRedirect(value: unknown) { const requested = String(value || "").trim(); const candidates = [requested, Deno.env.get("WEBAPP_URL"), WEBAPP_URL].filter(Boolean); for (const candidate of candidates) { try { const url = new URL(String(candidate)); if (url.protocol === "https:" && url.hostname === "appfmz.nl") { return url.origin; } } catch { // Ignore invalid redirect values. } } return WEBAPP_URL; } function isExistingUserError(error: unknown) { const message = String((error as { message?: string })?.message || error || "").toLowerCase(); return message.includes("already") || message.includes("registered") || message.includes("exists"); } serve(async (req) => { if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } if (req.method !== "POST") { return jsonResponse({ error: "Alleen POST is toegestaan." }, 405); } const supabaseUrl = Deno.env.get("SUPABASE_URL"); const anonKey = Deno.env.get("SUPABASE_ANON_KEY"); const serviceRoleKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY"); if (!supabaseUrl || !anonKey || !serviceRoleKey) { return jsonResponse({ error: "Supabase secrets ontbreken voor invite-client." }, 500); } const authorization = req.headers.get("Authorization") || ""; if (!authorization.startsWith("Bearer ")) { return jsonResponse({ error: "Log opnieuw in als trainer om een uitnodiging te sturen." }, 401); } const userClient = createClient(supabaseUrl, anonKey, { global: { headers: { Authorization: authorization } } }); const adminClient = createClient(supabaseUrl, serviceRoleKey, { auth: { autoRefreshToken: false, persistSession: false } }); const { data: userData, error: userError } = await userClient.auth.getUser(); if (userError || !userData?.user) { return jsonResponse({ error: "Trainer sessie kon niet worden gecontroleerd." }, 401); } const { data: trainerProfile, error: trainerError } = await userClient .from("profiles") .select("id, role, email, name") .eq("id", userData.user.id) .maybeSingle(); if (trainerError) { return jsonResponse({ error: trainerError.message }, 500); } if (!trainerProfile || trainerProfile.role !== "trainer") { return jsonResponse({ error: "Alleen trainers kunnen clienten uitnodigen." }, 403); } const body = await req.json().catch(() => ({})); const email = cleanEmail(body.email); const name = String(body.name || email.split("@")[0] || "Client").trim(); const clientId = String(body.clientId || "").trim(); const redirectTo = safeWebappRedirect(body.redirectTo); if (!email || !email.includes("@")) { return jsonResponse({ error: "Vul een geldig e-mailadres in." }, 400); } const userMetadata = { role: "client", trainer_id: trainerProfile.id, client_id: clientId, name }; const invite = await adminClient.auth.admin.inviteUserByEmail(email, { redirectTo, data: userMetadata }); if (!invite.error) { return jsonResponse({ ok: true, mode: "invite", redirectTo, message: "Uitnodiging verzonden." }); } if (!isExistingUserError(invite.error)) { return jsonResponse({ error: invite.error.message || "Uitnodiging kon niet worden verzonden." }, 400); } const recovery = await adminClient.auth.resetPasswordForEmail(email, { redirectTo }); if (recovery.error) { return jsonResponse({ error: recovery.error.message || "Uitnodiging kon niet opnieuw worden verzonden." }, 400); } return jsonResponse({ ok: true, mode: "password-reset-existing-user", redirectTo, message: "Uitnodiging verzonden." }); });
